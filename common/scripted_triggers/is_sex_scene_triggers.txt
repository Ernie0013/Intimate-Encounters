is_giving_partner_trigger = {
	OR = {
		AND = { # Futa sex, player is always giver
			carn_gender_can_impregnate_trigger = yes
			$PARTNER$ = { is_female = yes }
			is_ai = no
		}
		AND = { # Straight sex
			is_male = yes
			sex_opposite_of = $PARTNER$
		}
		AND = { # Gay sex
			sex_same_as = $PARTNER$
			is_ai = no
		}
	}
}

is_receiving_partner_trigger = {
	OR = {
		AND = { # Futa sex, AI is always receiver
			carn_is_futa_trigger = yes
			$PARTNER$ = { carn_gender_can_impregnate_trigger = yes }
			$PARTNER$ = { is_ai = no }
		}
		AND = { # Straight sex
			is_female = yes
			sex_opposite_of = $PARTNER$
		}
		AND = { # Gay sex
			sex_same_as = $PARTNER$
			$PARTNER$ = { is_ai = no }
		}
	}
}
